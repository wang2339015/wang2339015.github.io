git 常用命令

git init //初始化仓库

git add file.txt //添加要上传的文件

git commit -m “描述信息” //提交上传

git log //查看提交历史日志

git status //查看仓库状态

git diff file.txt //查看文件修改内容

git reset –hard commit_id //版本回朔 回到当初执行的命令

git reflog //查看命令历史

git checkout – file.txt //当你改乱了工作区某个文件的内容，想直接丢弃工作区的修改时 //rm file.txt 误删 ，能恢复

git rm file.txt //删除fiLe 文件

git remote add origin https://github.com/wang2339015/bb.git
//关联github上

git push origin master //本地提交到github上

第一次加 git push -u origin master

git clone https://github.com/wang2339015/bb.git
//克隆到本地

git branch //列出当前分支
